531  if (baud_vector.at(i) == iBaud) { baud_found = true; break; }
532  }
533  if (baud_found) {
534  port::error[1] = ok;
535  } else {
536  port::baud = 0; port::str_baud = "B0";
537  port::error[1] = err_baud_rate;
538  }
539  } else {
540  port::error[1] = err_baud_rate_not_set;
541  port::baud = 0; port::str_baud = "B0";
542  }
543  if (port::error[1].what().compare(NO_ERROR) == 0) {
544  port::baud = iBaud;
545  port::str_baud = "B" + to_string(iBaud);
546  if (cfsetispeed(&new_attr, baud) != 0) { // termios [SET INPUT baud] error occured
547  port::baud = 0; port::str_baud = "B0";
548  port::error[5] = err_can_t_set_termios_baud;
549  } else if (cfsetospeed(&new_attr, baud) != 0) { // termios [SET OUTPUT baud] error occured
550  port::baud = 0; port::str_baud = "B0";
551  port::error[5] = err_can_t_set_termios_baud;
552  } else { // termios [no error]
553  port::error[5] = ok;
554  }
555  } else {
556  port::baud = 0; port::str_baud = "B0";
557  }
